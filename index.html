class Boy:
def __init__(self, name):
self.name = name;
def loves(girlName):
print(self.name + ' loves ' + girlName);

class Girl:
def __init__(self, name):
self.name = name;
def misses(boyName):
print(self.name + ' misses ' + boyName);

boy = Boy('Lee');
girl = Girl('Rainbow');

// print out Lee loves Rainbow
boy.loves(girl.name);
// print out Rainbow misses Lee
girl.misses(boy.name);
